##############  Main Commands :

## Instantiating the Blueprint
resim call-function $package YtFair instantiate_ytfair

## exporting the component_address
$component="component_sim1qf07j4c825ee5x639cem366n57kvrgw983kpzcpmxczsc3ykew"

## creating a Content Creator Account as an NFT and also associating a Vault with it.
resim call-method $component make_cc_nft_cc_vault <Name of Youtube Channel>
resim call-method $component make_cc_nft_cc_vault KaustubhCooks

## creating a VideoNFT when video uploaded by the user.
resim call-method $component make_video_nft <Video Title> <Name of Youtube Channel> <Video Link From IPFS> 
resim call-method $component make_video_nft SchesuanSauce KaustubhCooks https://www.youtube.com/watch?v=XqZsoesa55w

## Recording View on a Video
resim call-method $component update_video_nft_views_byurl <Video Link From IPFS> 
resim call-method $component update_video_nft_views_byurl https://www.youtube.com/watch?v=XqZsoesa55w

## Recording Like on a Video
resim call-method $component update_video_nft_likes_byurl <Video Link From IPFS> 
resim call-method $component update_video_nft_likes_byurl https://www.youtube.com/watch?v=XqZsoesa55w

## Fetching INformation about a video
resim call-method $component fetch_video_details_and_update_view https://www.youtube.com/watch?v=XqZsoesa55w

## Randomly selecting a Video and Playing it i.e. returning its parameters to front end
resim call-method $component playvideo_for_video_feed

##Adding subscribers to the content creator
resim call-method $component update_cc_nft_subscribers_byurl <video url>
resim call-method $component update_cc_nft_subscribers_byurl "https://www.youtube.com/watch?v=XqZsoesa55w"

resim call-method $component update_cc_nft_subscribers_byccusername <ccnftusername>
resim call-method $component update_cc_nft_subscribers_byccusername KaustubhCooks

## Sending Money to a Content Creator's Vault
resim call-method $component deposit_cc_nft_cc_vault <Name of Youtube Channel> <Amount to be sent>
resim call-method $component deposit_cc_nft_cc_vault KaustubhCooks 150,resource_sim1qqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqqzqu57yag

## Content Creator withdrawing money from their respective vaults
resim call-method $component withdraw_from_cc_vault <Name of Youtube Channel> <Amount to be withdraw from vault>
resim call-method $component withdraw_from_cc_vault KaustubhCooks 70
